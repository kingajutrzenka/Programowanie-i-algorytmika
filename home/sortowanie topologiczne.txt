def sortowanie_topologiczne(sasiedzi):   #sasiedzi to po prostu lista sasiadow czyli w gruncie rzeczy slownik ktory podajemy jako argument 
    
    odwiedzone = set()
    wynik = []
    
    def odwiedz(wierzcholek):
        if wierzcholek in odwiedzone:
            return
        
        odwiedzone.add(wierzcholek)
        
        
        for sasiad in sasiedzi[wierzcholek]:
            odwiedz(sasiad)
        
        wynik.insert(0, wierzcholek)
    
    
    for wierzcholek in sasiedzi:
        odwiedz(wierzcholek)
    
    return wynik



# Przykład:
graf = {
    0: [],
    1: [],
    2: [3],
    3: [1],
    4: [0, 1],
    5: [0, 2]
}

print("Kolejność:", sortowanie_topologiczne(graf))
